Função get:
<script>
var get = function(url, callback){

			var xhr = new XMLHttpRequest();

			xhr.onreadystatechange = function(){

		if(xhr.readyState === 4){ //comunicacao com sucesso
			
		
			callback(xhr.responseText, xhr.status);

		}else{
	
			console.log(xhr.readyState);

		}
	};
	xhr.open('GET', url);
	xhr.send(null);
};
</sctript>
